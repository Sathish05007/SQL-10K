SQL Constraints

Primary Key

Ensures each record in a table is unique and not NULL.

CREATE TABLE students (
  id INT PRIMARY KEY,
  name VARCHAR(50)
);
Assignment: Create a "departments" table with dept_id as PRIMARY KEY.

Foreign Key

Ensures referential integrity between two tables.

CREATE TABLE orders (
  order_id INT PRIMARY KEY,
  customer_id INT,
  FOREIGN KEY(customer_id) REFERENCES customers(id)
);
Assignment: Create "enrollments" table with student_id as FOREIGN KEY referencing students(id).

Unique

Ensures all values in a column are unique.

CREATE TABLE users (
  email VARCHAR(100) UNIQUE,
  password VARCHAR(50)
);
Assignment: Create "employees" table with phone_number as UNIQUE.

SQL Clauses

ORDER BY

SELECT * FROM students ORDER BY age DESC;
Assignment: List employees ordered by salary (highest first).

DISTINCT

SELECT DISTINCT city FROM customers;
Assignment: Find unique course names from courses table.

LIMIT (MySQL) / TOP (SQL Server)

SELECT * FROM products LIMIT 5;  -- MySQL
SELECT TOP 5 * FROM products;    -- SQL Server
Assignment: Show the first 3 students from the students table.

GROUP BY + HAVING

SELECT department, COUNT(*) 
FROM employees 
GROUP BY department 
HAVING COUNT(*) > 5;
Assignment: Find departments having more than 10 employees.

SQL String Functions

UPPER()

SELECT UPPER(name) FROM students;
Assignment: Convert all product names to uppercase.

LOWER()

SELECT LOWER(email) FROM users;
Assignment: Convert customer names to lowercase.

CONCAT()

SELECT CONCAT(first_name, ' ', last_name) AS full_name FROM employees;
Assignment: Show full names of students.

SUBSTRING()

SELECT SUBSTRING(name, 1, 3) FROM students;
Assignment: Extract first 4 letters of each product name.

SQL Date Functions

NOW()

SELECT NOW();
Assignment: Print current system date and time.

CURDATE()

SELECT CURDATE();
Assignment: Get only the current date.

YEAR(), MONTH(), DAY()

SELECT YEAR(order_date), MONTH(order_date), DAY(order_date) FROM orders;
Assignment: Extract year of joining for each employee.

DATEDIFF()

SELECT DATEDIFF(end_date, start_date) FROM projects;
Assignment: Find number of days between order_date and delivery_date.
