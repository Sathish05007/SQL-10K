SQL Operators

= (Equal)

SELECT * FROM table WHERE column = value;
Example: SELECT * FROM students WHERE age = 18;
Assignment: Find all employees whose department_id = 2.

< (Less Than)

SELECT * FROM table WHERE column < value;
Example: SELECT * FROM products WHERE price < 100;
Assignment: List all products cheaper than 500.

> (Greater Than)

SELECT * FROM table WHERE column > value;
Example: SELECT * FROM employees WHERE salary > 50000;
Assignment: Show all employees earning more than 60,000.

LIKE

SELECT * FROM table WHERE column LIKE pattern;
Example: SELECT * FROM students WHERE name LIKE 'A%';
Assignment: Find students whose name starts with 'S'.

BETWEEN

SELECT * FROM table WHERE column BETWEEN value1 AND value2;
Example: SELECT * FROM orders WHERE amount BETWEEN 100 AND 500;
Assignment: List all products with price between 200 and 800.

IN

SELECT * FROM table WHERE column IN (value1, value2, ...);
Example: SELECT * FROM employees WHERE department IN ('HR','IT');
Assignment: Find all students in classes 1, 3, and 5.

SQL Aggregation Functions

COUNT()

SELECT COUNT(column) FROM table;
Example: SELECT COUNT(*) FROM students;
Assignment: Count how many employees are in the company.

SUM()

SELECT SUM(column) FROM table;
Example: SELECT SUM(salary) FROM employees;
Assignment: Find total sales amount from orders table.

AVG()

SELECT AVG(column) FROM table;
Example: SELECT AVG(price) FROM products;
Assignment: Find the average age of students.

MIN()

SELECT MIN(column) FROM table;
Example: SELECT MIN(salary) FROM employees;
Assignment: Find the lowest product price.

MAX()

SELECT MAX(column) FROM table;
Example: SELECT MAX(marks) FROM students;
Assignment: Find the highest order amount.
